class Solution {
    public int subarraySum(int[] nums, int t) {
        int len=0;
        int n=nums.length;
        for(int i=0;i<n;i++){
            int sum=0;
            for(int j=i;j<n;j++){
                    sum+=nums[j];
                    if(sum==t){
                        len=Math.max(len,j-i+1);
                }
            }
        }
    return len;
}
}
