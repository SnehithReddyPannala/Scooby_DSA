class Solution {
    void selectionSort(int[] arr) {
      
    int n = arr.length;
    for (int i = n - 1; i > 0; i--) {
        int max = 0;
        for (int j = 0; j <= i; j++) {
            if (arr[max] < arr[j]) {
                max = j;
            }
        }
        int temp = arr[i];
        arr[i] = arr[max];
        arr[max] = temp;
    }
     
}

    }
