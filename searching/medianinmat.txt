class Solution {
    public int median(int[][] mat) {
        int r=mat.length;
        int c=mat[0].length;
     List<Integer> list = new ArrayList<>();
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++) {
            list.add(mat[i][j]);
        }
    }
    Collections.sort(list);
    int n = list.size();
    return list.get(n / 2);
        
    }
}
